// Exemplo 1: Declaração simples de variáveis
int x;
int y;
int z;

// Atribuição e inicialização
x = 10;
y = 20;
z = x + y;

// Declaração não usada (aviso)
int w;

// Uso de variável não inicializada (aviso)
int a;
int b;
b = a + 5;

// Estrutura condicional
se (x > 5) entao
    y = y + 1;
end

// Laço enquanto
enquanto (z < 100) entao
    z = z + x;
end

// Laço para
para (int i = 0; i < 10; i = i + 1) faca
    z = z + 1;
end

// Saída
escreva (x);
escreva (y);
escreva (z);
escreva (w);
